Comando para instalação

Recomenda-se, previamente, atualização da distro, com 'sudo apt update && sudo apt full-upgrade'.

sudo dpkg --add-architecture i386 && sudo apt update && sudo apt install -y git idle python3-dev python3-venv python3-pip python3-pygame wget curl gnupg gnupg-agent htop vim figlet neofetch tree net-tools telnet openssh-server flatpak gnome-software-plugin-flatpak default-jdk vagrant nodejs npm composer obs-studio kdenlive pass software-properties-common neovim tmux ranger ca-certificates hollywood cowsay locate gparted openssl shotwell mugshot ufw gufw aptitude simple-scan galculator bluez blueman gcc make linux-headers-$(uname -r) dkms qemu-kvm libvirt-clients libvirt-daemon-system bridge-utils virt-manager libguestfs-tools ffmpeg libc6:i386 libncurses5:i386 libstdc++6:i386 lib32z1 libbz2-1.0:i386 android-tools-adb android-tools-fastboot ansible && sudo updatedb && sudo systemctl enable bluetooth --now && sudo pip3 install django djangorestframework markdown django-filter --break-system-packages && sudo npm install -g @angular/cli && sudo npm install -g @vue/cli && curl https://cli-assets.heroku.com/install.sh | sh && flatpak remote-add --if-not-exists flathub https://dl.flathub.org/repo/flathub.flatpakrepo && sudo install -m 0755 -d /etc/apt/keyrings && curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg && sudo chmod a+r /etc/apt/keyrings/docker.gpg && echo \
    "deb [arch="$(dpkg --print-architecture)" signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/debian \
    "$(. /etc/os-release && echo "$VERSION_CODENAME")" stable" | \
    sudo tee /etc/apt/sources.list.d/docker.list > /dev/null && sudo apt update && sudo apt install -y docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin && sudo groupadd docker && sudo usermod -aG docker $USER && newgrp docker && wget http://archive.ubuntu.com/ubuntu/pool/main/o/openssl/libssl1.1_1.1.1f-1ubuntu2_amd64.deb && sudo dpkg -i libssl1.1_1.1.1f-1ubuntu2_amd64.deb && curl -fsSL https://pgp.mongodb.com/server-7.0.asc | \
     sudo gpg -o /usr/share/keyrings/mongodb-server-7.0.gpg \
     --dearmor && echo "deb [ signed-by=/usr/share/keyrings/mongodb-server-7.0.gpg ] http://repo.mongodb.org/apt/debian bullseye/mongodb-org/7.0 main" | sudo tee /etc/apt/sources.list.d/mongodb-org-7.0.list && sudo apt update && sudo apt install -y mongodb-org && rm libssl1.1_1.1.1f-1ubuntu2_amd64.deb
